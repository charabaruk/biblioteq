128 bool haveMemberChanges(QString &str)
129 {
130 QStringList list;
131
132 if(m_memberProperties.value("city") != m_userinfo.city->text().trimmed())
133 list << "city";
134
135 if(m_memberProperties.value("comments") !=
136 m_userinfo.comments->toPlainText().trimmed())
137 list << "comments";
138
139 if(m_memberProperties.value("dob") !=
140 m_userinfo.dob->date().toString(Qt::ISODate))
141 list << "dob";
142
143 if(m_memberProperties.value("email") != m_userinfo.email->text())
144 list << "email";
145
146 if(m_memberProperties.value("expiration_date") !=
147 m_userinfo.expirationdate->date().toString(Qt::ISODate))
148 list << "expiration_date";
149
150 if(m_memberProperties.value("first_name") !=
151 m_userinfo.firstName->text().trimmed())
152 list << "first_name";
153
154 if(m_memberProperties.value("general_registration_number") !=
155 m_userinfo.generalregistrationnumber->text().trimmed())
156 list << "general_registration_number";
157
158 if(m_memberProperties.value("last_name") !=
159 m_userinfo.lastName->text().trimmed())
160 list << "last_name";
161
162 if(m_memberProperties.value("maximum_reserved_books") !=
163 m_userinfo.maximum_reserved_books->text())
164 list << "maximum_reserved_books";
165
166 if(m_memberProperties.value("memberclass") !=
167 m_userinfo.memberclass->text().trimmed())
168 list << "memberclass";
169
170 if(m_memberProperties.value("membership_fees") !=
171 m_userinfo.membershipfees->text())
172 list << "membership_fees";
173
174 if(m_memberProperties.value("membersince") !=
175 m_userinfo.membersince->date().toString(Qt::ISODate))
176 list << "membersince";
177
178 if(m_memberProperties.value("middle_init") !=
179 m_userinfo.middle->text().trimmed())
180 list << "middle_init";
181
182 if(m_memberProperties.value("overdue_fees") !=
183 m_userinfo.overduefees->text())
184 list << "overdue_fees";
185
186 if(m_memberProperties.value("sex") != m_userinfo.sex->currentText())
187 list << "sex";
188
189 if(m_memberProperties.value("state_abbr") !=
190 m_userinfo.state->currentText())
191 list << "state_abbr";
192
193 if(m_memberProperties.value("street") !=
194 m_userinfo.street->text().trimmed())
195 list << "street";
196
197 if(m_memberProperties.value("telephone_num") !=
198 m_userinfo.telephoneNumber->text())
199 list << "telephone_num";
200
201 if(m_memberProperties.value("zip") != m_userinfo.zip->text())
202 list << "zip";
203
204 std::sort(list.begin(), list.end());
205 str.clear();
206
207 for(int i = 0; i < list.size(); i++)
208 str += list.at(i) + ", ";
209
210 if(!str.isEmpty())
211 {
212 str.prepend("(");
213 str = str.mid(0, str.length() - 2);
214 str.append(")");
215 }
216
217 return !str.isEmpty();
218 }

240 private slots:
241 void done(int result)
242 {
243 QString str("");
244
245 if(haveMemberChanges(str))
246 if(QMessageBox::
247 question(this,
248 tr("BiblioteQ: Question"),
249 tr("Your changes have not been committed. "
250 "Continue closing?\n%1").arg(str),
251 QMessageBox::No | QMessageBox::Yes,
252 QMessageBox::No) == QMessageBox::No)
253 {
254 QApplication::processEvents();
255 return;
256 }
257
258 QDialog::done(result);
259 }

261 void slotExpirationDate(void)
262 {
263 auto action = qobject_cast<QAction *> (sender());
264
265 if(!action)
266 return;
267
268 auto date(QDate::currentDate());
269
270 date.setDate(date.year(), 12, 31);
271 m_userinfo.expirationdate->setDate(date.addYears(action->data().toInt()));
272 }
